\begin{minted}{typescript}
  private _wrapWebWorkerPromise(
    id: string,
    promise: Promise<any>,
    type: Type<any> | SerializerTypes
  ): void {
    promise.then((result: any) => {
      this._sink.emit({
        type: 'result',
        value: this._serializer.serialize(result, type),
        id: id,
      });
    });
  }
\end{minted}
